## 📋 VERİTABANI DEĞİŞİKLİKLERİ

### ClientNotification (İletişim Logu)
```prisma
model ClientNotification {
  id          String   @id @default(cuid())
  tenantId    String
  clientId    String
  client      Client   @relation(fields: [clientId], references: [id])
  caseId      String?
  case        Case?    @relation(fields: [caseId], references: [id])
  channel     String   // EMAIL, SMS, WHATSAPP
  type        String   // MASRAF_ISTEK, GENEL_BILGILENDIRME, RAPOR
  subject     String?
  body        String
  status      String   @default("PENDING") // PENDING, SENT, FAILED
  sentAt      DateTime?
  sentById    String
  sentBy      User     @relation(fields: [sentById], references: [id])
  errorMessage String?
  createdAt   DateTime @default(now())
}
```

### ExpenseRequest (Masraf Talebi)
```prisma
model ExpenseRequest {
  id          String   @id @default(cuid())
  tenantId    String
  caseId      String
  case        Case     @relation(fields: [caseId], references: [id])
  clientId    String
  client      Client   @relation(fields: [clientId], references: [id])
  items       Json     // [{description, amount}]
  totalAmount Decimal
  status      String   @default("PENDING") // PENDING, APPROVED, REJECTED, PAID
  sentAt      DateTime?
  sentVia     String?  // EMAIL, SMS
  respondedAt DateTime?
  paidAt      DateTime?
  notes       String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}
```

### Client Güncellemesi
```prisma
model Client {
  // ... mevcut alanlar
  portalEnabled Boolean @default(false)
  // ... relations
  notifications ClientNotification[]
  expenseRequests ExpenseRequest[]
}
```

### User Güncellemesi
```prisma
model User {
  // ... mevcut alanlar
  globalRole String @default("STAFF") // ADMIN, LAWYER, STAFF, CLIENT
  clientId   String? // CLIENT rolü için
  client     Client? @relation(fields: [clientId], references: [id])
}
```
